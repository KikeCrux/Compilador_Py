program {
    int m, n;
    float x, y;
    bool z;
    z = true;
    m = 15;
    n = 25;
    x = 2.5;
    y = 1.5;
    do {
        if(m == n and x <= y) then {
            z = false;
        } else {
            m = m + 3;
            x = y * x + n;
            n = n - 2;
        }
        fi
        while(x >= 5 or m == n){
            write m;
            x = x / 1.1;
            n = n + 1;
        }
    } until(z == false);
}


program {

    flag = false;
    i = 8;
    j = 12;
    p = 4.4;
    q = 2.2;
    do {
        if(i != j and j > 5) then {
            flag = true;
        } else {
            i = i + 2;
            p = p - q;
            j = j - 1;
        }
        fi
        while(p <= 10 or i != j){
            write i;
            p = p * 1.5;
            q = q + 0.5;
        }
    } until(flag == true);
}
